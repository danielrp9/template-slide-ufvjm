% PACOTES E FONTES
\usepackage[brazil]{babel}
\usepackage[T1]{fontenc}

% DESIGN SYSTEM (UFVJM)
\definecolor{ufvjmBlue}{HTML}{005EB8}
\definecolor{ufvjmGreen}{HTML}{A7CF3A}
\definecolor{ufvjmDark}{HTML}{1E293B} 
\definecolor{ufvjmSlate}{HTML}{64748B} 

% CONFIGURAÇÕES DE LAYOUT
\setbeamertemplate{navigation symbols}{}
\setbeamercolor{background canvas}{bg=white}
\setbeamersize{text margin left=1.5cm, text margin right=1.5cm}

% ESTILO DE LISTAS
\setbeamercolor{itemize item}{fg=ufvjmBlue}
\setbeamercolor{itemize subitem}{fg=ufvjmGreen}
\setbeamertemplate{itemize item}{\small\raise0.5pt\hbox{\begin{tikzpicture} \fill[ufvjmBlue] (0,0) -- (0.15,0) -- (0.2,0.15) -- (0.05,0.15) -- cycle; \end{tikzpicture}}}
\setbeamercolor{section in toc}{fg=ufvjmDark}

% VARIÁVEIS DE CONTROLE
\newcommand{\titulo}[1]{\title{#1}\def\inserttitulo{#1}} 
\newcommand{\tipoTrabalho}[1]{\def\inserttipo{#1}}
\newcommand{\estudantes}[1]{\def\insertestudantes{#1}}
\newcommand{\caminhoLogo}[1]{\def\insertlogo{#1}}

% COMANDO PARA SLIDE DE TRANSIÇÃO
\newcommand{\secao}[1]{
    \section{#1}
    \begin{frame}[plain]
        \begin{tikzpicture}[remember picture,overlay]
            \fill[white] (current page.north west) rectangle (current page.south east);
            \fill[ufvjmBlue] (current page.north west) rectangle ([xshift=0.5cm]current page.south west);
            \node[anchor=center] at (current page.center) {
                \begin{minipage}{0.8\paperwidth}
                    \centering
                    {\color{ufvjmSlate}\fontsize{10}{10}\selectfont\uppercase{\textbf{Tópico}}}\\[0.4cm]
                    {\color{ufvjmDark}\fontsize{24}{28}\selectfont\textbf{#1}}
                \end{minipage}
            };
        \end{tikzpicture}
    \end{frame}
}

% HEADER DAS PÁGINAS INTERNAS
\setbeamertemplate{frametitle}{
    \nointerlineskip
    \begin{tikzpicture}[remember picture,overlay]
        \shade[left color=ufvjmBlue, right color=ufvjmGreen] (current page.north west) rectangle ([yshift=-1cm]current page.north east);
        \node[anchor=west, xshift=1.5cm, yshift=-0.5cm, inner sep=0pt] at (current page.north west) {
            \begin{minipage}{0.75\paperwidth}
                \raggedright\fontsize{14}{14}\selectfont\color{white}\textbf{\insertframetitle}
            \end{minipage}
        };
    \end{tikzpicture}
    \vspace{1.2cm}
}

% RODAPÉ DAS PÁGINAS INTERNAS (SEM LOGO REPETIDA)
\setbeamertemplate{footline}{
    \begin{tikzpicture}[remember picture,overlay]
        \draw[gray!15, line width=0.5pt] ([xshift=1.5cm, yshift=0.8cm]current page.south west) -- ([xshift=-1.5cm, yshift=0.8cm]current page.south east);
        \node[anchor=south west, xshift=1.5cm, yshift=0.35cm] at (current page.south west) {
            \fontsize{5}{5}\selectfont\color{ufvjmSlate}\textbf{\insertshorttitle}
        };
        \node[anchor=south east, xshift=-1.5cm, yshift=0.35cm] at (current page.south east) {
            \fontsize{6}{6}\selectfont\color{ufvjmSlate}\textbf{\insertframenumber \ / \inserttotalframenumber}
        };
    \end{tikzpicture}
}
