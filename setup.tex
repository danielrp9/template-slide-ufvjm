% HEADER DAS PÁGINAS INTERNAS
\setbeamertemplate{frametitle}{
    \nointerlineskip
    \begin{tikzpicture}[remember picture,overlay]
        \shade[left color=ufvjmBlue, right color=ufvjmGreen] (current page.north west) rectangle ([yshift=-1cm]current page.north east);
        \node[anchor=west, xshift=1.5cm, yshift=-0.5cm, inner sep=0pt] at (current page.north west) {
            \begin{minipage}{0.75\paperwidth}
                \raggedright\fontsize{14}{14}\selectfont\color{white}\textbf{\insertframetitle}
            \end{minipage}
        };
    \end{tikzpicture}
    \vspace{1.2cm}
}

% RODAPÉ
\setbeamertemplate{footline}{
    \begin{tikzpicture}[remember picture,overlay]
        \draw[gray!15, line width=0.5pt] (current page.south west) ++(1.5cm,0.8cm) -- ([xshift=-1.5cm, yshift=0.8cm]current page.south east);
        \node[anchor=south west, xshift=1.5cm, yshift=0.35cm] at (current page.south west) {
            \fontsize{5}{5}\selectfont\color{ufvjmSlate}\uppercase{UFVJM \ | \ \insertshorttitle}
        };
        \node[anchor=south east, xshift=-1.5cm, yshift=0.35cm] at (current page.south east) {
            \fontsize{6}{6}\selectfont\color{ufvjmSlate}\textbf{\insertframenumber \ / \inserttotalframenumber}
        };
    \end{tikzpicture}
}

\newcommand{\gerarCapa}{
    \begin{frame}[plain]
        \hypersetup{pdfborder={0 0 0}, hidelinks}
        \begin{tikzpicture}[remember picture,overlay]
            \shade[top color=ufvjmBlue, bottom color=ufvjmGreen] (current page.north west) rectangle ([xshift=0.15cm]current page.south west);
            \shade[top color=ufvjmGreen, bottom color=ufvjmBlue] ([xshift=-0.1cm]current page.north east) rectangle (current page.south east);
            
            \node[anchor=north] (logo) at ($(current page.north) + (0,-0.8cm)$) {
                \includegraphics[height=0.8cm]{\insertlogo}
            };
            
            \draw[gray!30, line width=0.5pt] ([xshift=1.5cm]current page.north west) ++(0,-1.2cm) -- ([xshift=-1cm]logo.west);
            \draw[gray!30, line width=0.5pt] ([xshift=1cm]logo.east) -- ([xshift=-1.5cm, yshift=-1.2cm]current page.north east);

            \node[anchor=center, inner sep=0pt] at ($(current page.center) + (0,0.1cm)$) {
                \begin{minipage}{0.85\paperwidth}
                    \centering
                    {\color{ufvjmBlue}\fontsize{8}{8}\selectfont\uppercase{\textbf{\inserttipo}}}\\[0.5cm] 
                    {\fontsize{28}{32}\selectfont\color{ufvjmDark}\textbf{\inserttitulo}}
                \end{minipage}
            };

            \node[anchor=south west, xshift=1.5cm, yshift=1.2cm] at (current page.south west) {
                \begin{minipage}{0.85\paperwidth}
                    \raggedright
                    {\fontsize{6}{6}\selectfont\color{ufvjmSlate}\uppercase{Discentes:}}\\[0.15cm]
                    {\small\color{ufvjmDark}\textbf{\insertestudantes}}
                \end{minipage}
            };
        \end{tikzpicture}
    \end{frame}
}
